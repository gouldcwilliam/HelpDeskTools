@ECHO OFF

	SET A=%1
	SET S=%2

:BEGIN
	IF %A%==start CALL:START 		
	IF %A%==stop CALL:STOP
	IF %A%==restart (
		CALL:STOP
		CALL:START
	)
GOTO:DONE

:STOP
	IF %S%==credit  CALL:cSTOP 
	IF %S%==sql  CALL:sSTOP
	IF %S%==pca CALL:pSTOP 
	IF %S%==all ( 
		CALL:cSTOP
		CALL:sSTOP
		CALL:pSTOP
	)
GOTO:EOF
		
:START
	IF %S%==credit  CALL:cSTART 
	IF %S%==sql  CALL:sSTART 
	IF %S%==pca CALL:pSTART
	IF %S%==all ( 
		CALL:cSTART
		CALL:sSTART
		CALL:pSTART
	)
GOTO:EOF
	
:cSTART
	NET START "RIBROKER"
	NET START "CDCA MULTI CLIENT"
	NET START "NFM CLIENT SERVICE"
	NET START "NTM JAVA CLIENT"
	NET START "TRANSNET"
GOTO:EOF
	
	
:cSTOP
	NET STOP "CDCA MULTI CLIENT"
	NET STOP "RIBROKER"
GOTO:EOF

:sSTART
	NET START "MSSQLserver"
	NET START "XPRESS SERVER"
	NET START "NFM CLIENT SERVICE"
	NET START "NTM JAVA CLIENT"
	NET START "TRANSNET"
GOTO:EOF

:sSTOP
	NET STOP "XPRESS SERVER"
	NET STOP "MSSQLSERVER"
	NET STOP "TRANSNET"
GOTO:EOF

:pSTOP
	NET STOP "awhost32"
GOTO:EOF

:pSTART
	NET START "awhost32"
GOTO:EOF
	
:KILL
	TASKKILL /IM "POSW.EXE" /F
GOTO:EOF

:WAIT
	PING 127.0.0.1 -n 2 > NUL
GOTO:EOF

:DONE
